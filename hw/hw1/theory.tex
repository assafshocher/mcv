\noindent
Recall: $(f \conv g)(t) = \int_{-\infty}^{\infty} f(\tau)\, g(t - \tau)\, d\tau$.
\;A system $T$ is \textbf{LTI} (linear time-invariant) if
$T\{\alpha f + \beta g\} = \alpha\, T\{f\} + \beta\, T\{g\}$ and
$T\{f(t-t_0)\} = (Tf)(t-t_0)$.

\vspace{8pt}
\noindent\textbf{Prove:}

\begin{enumerate}[leftmargin=2em, itemsep=6pt]

  \item \textbf{Convolution is commutative:}\quad $f \conv g \;=\; g \conv f$

  \item \textbf{Convolution is associative:}\quad $(f \conv g) \conv h \;=\; f \conv (g \conv h)$

  \item \textbf{Convolution is distributive:}\quad $f \conv (g + h) \;=\; f \conv g \;+\; f \conv h$

  \item \textbf{Delta is the identity:}\quad $f \conv \delta \;=\; f$

  \item \textbf{Convolution commutes with differentiation:}
  \[\frac{d}{dt}(f \conv g) \;=\; \frac{df}{dt} \conv g \;=\; f \conv \frac{dg}{dt}\]

  \item \textbf{A system is LTI if and only if it can be expressed by a convolution:}\quad $T$ is LTI $\Leftrightarrow$ $T\{f\} = f \conv h$ for some kernel $h$.

  \textcolor{gray}{\small [Hint: $\Leftarrow$ is easy.\; $\Rightarrow$\, express $f$ using $\delta$]}

  \item \textbf{Gaussians convolve into Gaussians:}
  \[ g_{\sigma_1} \conv g_{\sigma_2} \;=\; g_{\sqrt{\sigma_1^2 + \sigma_2^2}} \,,\quad \text{where}\quad g_\sigma(t) = \tfrac{1}{\sigma\sqrt{2\pi}}\exp\!\left(-\tfrac{t^2}{2\sigma^2}\right) \]

  \textcolor{gray}{\small [Hint: complete the square]}

  \item \textbf{Repeated convolution converges to a Gaussian.}

  Let $X_1, X_2$ be two independent random variables with PDFs $f_1, f_2$.
  Define $S_2 = X_1 + X_2$.

  \begin{enumerate}[label=(\alph*), itemsep=4pt]

    \item Write down the joint PDF of $(X_1, X_2)$.
    What does independence let you do?

    \item Express the CDF $F_{S_2}(s) = P(S_2 \le s)$ as a double integral
    over the region $\{x_1 + x_2 \le s\}$, using the joint PDF from~(a).

    \item Differentiate $F_{S_2}(s)$ with respect to $s$ to obtain the PDF $f_{S_2}(s)$.
    What familiar operation appears?

    \item Now let $X_1, \ldots, X_n$ be IID with PDF $f$, mean $\mu$,
    and finite variance $\sigma^2$. Let $S_n = X_1 + \cdots + X_n$.
    Use the result from~(c) to express $f_{S_n}$ in terms of $f$.

    \item State the Central Limit Theorem (CLT) for $S_n$.
    What does it say about $f_{S_n}$ for large $n$?

    \textcolor{gray}{\small
    Recall: the CLT states that $(S_n - n\mu)/(\sigma\sqrt{n}) \to \mathcal{N}(0,1)$
    in distribution as $n \to \infty$.
    See e.g.\ \texttt{en.wikipedia.org/wiki/Central\_limit\_theorem}.}

    \item Combine (d) and (e): what does this tell you about
    repeatedly convolving a nonnegative, normalized function with itself?

  \end{enumerate}

\end{enumerate}
